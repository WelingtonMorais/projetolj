CREATE DATABASE db_estoque;

USE db_estoque;

CREATE TABLE tb_produto(id BIGINT AUTO_INCREMENT NOT NULL,nome NVARCHAR(100) NOT NULL,quantidade BIGINT NOT NULL,PRIMARY KEY(id));

CREATE TABLE tb_pessoa(id BIGINT AUTO_INCREMENT NOT NULL,cpf BIGINT NOT NULL,nome NVARCHAR(100) NOT NULL,data INT NOT NULL,telefone BIGINT NOT NULL,email NVARCHAR(100) NOT NULL,PRIMARY KEY(id));

CREATE TABLE tb_categoria(id BIGINT AUTO_INCREMENT NOT NULL,produto NVARCHAR(100) NOT NULL,categoria NVARCHAR(100) NOT NULL,PRIMARY KEY(id));

SHOW TABLES;
